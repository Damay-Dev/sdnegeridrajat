SINKRONISASI GOOGLE SHEETS & AI KEY

💡 Blueprint Google Apps Script Gunakan kode Apps Script berikut di Google Spreadsheet Anda untuk mengaktifkan real-time chatbot dan perekaman survei SKM: 
function doPost(e) {
  var params = e.parameter;
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; // Menggunakan sheet pertama secara mutlak agar tidak salah baca tab
  
  // Jika aksi adalah kirim data SKM
  if (params.name) {
    sheet.appendRow([params.date, params.name, params.rating, params.comment]);
    return ContentService.createTextOutput("SUCCESS")
      .setMimeType(ContentService.MimeType.TEXT);
  }
}

function doGet(e) {
  var action = e.parameter.action;
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; // Menggunakan sheet pertama secara mutlak agar tidak salah baca tab
  
  // Mengambil ulasan SKM real-time
  if (action === "getSkm") {
    // MENGGUNAKAN getDisplayValues() agar format tanggal dan teks di website tidak berubah/berantakan
    var data = sheet.getDataRange().getDisplayValues(); 
    var results = [];
    
    // Skip baris pertama (header)
    for (var i = 1; i < data.length; i++) {
      results.push({
        id: i,
        date: data[i][0] ? data[i][0] : "",
        name: data[i][1] ? data[i][1] : "",
        rating: data[i][2] ? data[i][2] : "",
        comment: data[i][3] ? data[i][3] : ""
      });
    }
    
    // Return dalam format JSON dengan header CORS lengkap secara otomatis oleh Google
    return ContentService.createTextOutput(JSON.stringify(results))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
